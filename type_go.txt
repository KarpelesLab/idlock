package idlock

import "sync"

type TypeNameLock struct {
	mp map[typename]bool
	lk sync.Mutex
	cd *sync.Cond
}

func NewTypeName() *TypeNameLock {
	lk := &TypeNameLock{
		mp: make(map[typename]bool),
	}
	lk.cd = sync.NewCond(&lk.lk)

	return lk
}

func (lk *TypeNameLock) Lock(i ...typename) {
	var f bool

	if len(i) == 0 {
		return
	}

	lk.lk.Lock()
	for {
		for _, n := range i {
			if _, f = lk.mp[n]; f {
				break
			}
		}
		if !f {
			// mark typenames as locked
			for _, n := range i {
				lk.mp[n] = true
			}
			lk.lk.Unlock()
			return
		}

		// wait for an update
		lk.cd.Wait()
	}
}

func (lk *TypeNameLock) Unlock(i ...typename) {
	if len(i) == 0 {
		return
	}

	lk.lk.Lock()

	for _, n := range i {
		delete(lk.mp, n)
	}

	lk.cd.Broadcast()
	lk.lk.Unlock()
}
